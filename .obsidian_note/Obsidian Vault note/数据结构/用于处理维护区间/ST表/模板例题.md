void solve() {
    int m,n;
    cin >> m >> n;
    int max_j = log2(m) + 1;
    vector<vector<int>> st(m,vector<int>(max_j));
    for(int i = 0;i < m;i++) {
        cin >> st[i][0];
    }
    for(int j = 1;j < max_j;j++) {
        for(int i = 0;i + (1 << j) <= m;i++) {
            st[i][j] = min(st[i][j - 1],st[i + (1 << (j - 1))][j - 1]);
        }
    }
    while(n--) {
        int a,b;
        cin >> a >> b;
        int len = b - a + 1;
        a--;
        b--;

## 例题2：
[3691. 最大子数组总值 II](https://leetcode.cn/problems/maximum-total-subarray-value-ii/)
**示例 2 的 nums=[4,2,5,1]，把所有连续子数组的值（极差）算出来，可以得到一个矩阵 M，其中 M[l,r]表示区间 [l,r] 的值（mx - mi）;**
![配图](../../../../截图文件/Pasted%20image%2020260610201408.png)
这里会发现矩阵中的每一行都是递增的；
- **则在第一次的最大值中的候选位置中，一定是存在于最后一行；
- **当我们取了当前最大值区间时，之后就不可以重复使用了，这时最大值中的候选候选位置，则再排除上一层中的已选的最大值区间位置  + 与当前最大区间[l,r]的前一个位置[l,r - 1]
- **这里贪心的数据维护可以用最大堆来完成；同时这里可以作点简化，当前的区间值如果为0时候，其实就不再需要入队其左侧区间，因为这是左侧的数值一定为0。这样当如果k比较多的情况下就不需要遍历一个其右侧状态区间值为0的区间从而适当简化；
同时这里需要任意区间的max,min值，还是个静态问题，可以采用ST表

**代码如下**：
```c++
class Solution {
public:
    vector<vector<int>> st1,st2;

    void init(vector<int>& nums) {
        size_t n = nums.size();
        int m = bit_width(n);
        st1.resize(n, vector<int>(m));
        st2.resize(n, vector<int>(m));

        for(int i = 0;i < n;i++) {
            st1[i][0] = st2[i][0] = nums[i];
        }
        for(int j = 1;j < m;j++) {
            for(int i = 0;i + (1 << j) <= n;i++) {
                st1[i][j] = min(st1[i][j - 1],st1[i + (1 << (j - 1))][j - 1]);
                st2[i][j] = max(st2[i][j - 1],st2[i + (1 << (j - 1))][j - 1]);
            }
        }
    }
    //[)

    int query(int l,int r) {
        //if(l >= r) return 0;
        int k = bit_width((size_t)r - l) - 1;
        int mx = max(st2[l][k],st2[r - (1 << k)][k]);
        int mi = min(st1[l][k],st1[r - (1 << k)][k]);
        return mx - mi;
    }

    long long maxTotalValue(vector<int>& nums, int k) {
        int n = nums.size();
        init(nums);
        priority_queue<tuple<int,int,int>> pq;
        for(int i = 0;i < n;i++) {
            pq.emplace(query(i,n),i,n);//[)
        }

        long long ans = 0;
        while(k-- && !pq.empty()) {
            auto [add,l,r] = pq.top();
            if(!add) continue;
            pq.pop();
            ans += add;
            pq.emplace(query(l,r - 1),l,r - 1);
        }
        return ans;
    }
};
```
