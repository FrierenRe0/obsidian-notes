## 2.（还原树）给定二叉树前，中，后序还原二叉树的层序

**还原树（结构体）模板【给定后序 + 中序 - 还原整颗树！】**
**关键是理解记忆 build 函数**
1. 后序的最后一个是根。即a[pr] 是根，我们先将他拿出来，创一个新节点放进去
2. 遍历中序数组，找到我们需要的根的位置 k
3. 计算中序数组中，根左边的节点数量 cnt。 cnt = k - il
4. 对于中序，我们将根左边范围内的 传到  左子树中[il, k - 1]
5.              ， 我们将根右边范围 的传到 右子树中[k + 1, ir]
6. 对于后序，我们将给定的后序 范围内的前cnt个数 传到左子树中
7.               ，我们将                               剩下的数 传到右子树中
8. 最后别忘了一开始的递归边界：pl > pr


**结构体写法**：
```c++
#include<bits/stdc++.h>
using namespace std;

struct node {
    int val;
    node* left, *right;

    node(int v) : val(v), left(NULL), right(NULL) {}
};

vector<int> post, in;
int n;

// 递归建树：后序[pl,pr]  中序[il,ir]
node * build(int pl, int pr, int il, int ir) {
    if (pl > pr) return NULL;

    int root_val = post[pr];
    node * root = new node(root_val);

    int k;
    for (k = il; k <= ir; k++) {
        if (in[k] == root_val) break;
    }

    int cnt = k - il;

    root->left = build(pl, pl + cnt - 1, il, k - 1);
    root->right = build(pl + cnt, pr - 1, k + 1, ir);

    return root;
}


void ap() {
    cin >> n;
    post.resize(n);
    in.resize(n);
    for (int i = 0; i < n; i++) cin >> post[i];
    for (int i = 0; i < n; i++) cin >> in[i];

    node *root = build(0, n - 1, 0, n - 1);
    // 就可以得到想要的遍历操作了！

    return;
}

signed main () {
    int t; cin >> t;
    while (t--) ap();

    return 0;
}
```

- **后序遍历**：左子树 → 右子树 → 根
- **层序遍历**：从上到下、从左到右一层一层输出（用**队列**实现）
![配图](../../../截图文件/Pasted%20image%2020260414111412.png)

# 解题思路：
## 从 后序 + 中序 还原二叉树的**核心规律**

1. **后序遍历的最后一个元素，一定是整棵树的根节点**
2. 在**中序遍历**里找到这个根：

    - 根左边 = **左子树**所有节点
    - 根右边 = **右子树**所有节点

3. 用**递归**重复这个过程：

    - 找左子树的根
    - 找右子树的根
    - 直到所有节点都建好

**后序**：`2 3 1 5 7 6 ④`
**中序**：`1 2 3 ④ 5 6 7`

## 第一步：整棵树的根

后序：`2 3 1 5 7 6 ④`

最后一个是 **4**

→ **整棵树根是 4**

看**中序**：

`1 2 3 ④ 5 6 7`

- 4 左边：`1 2 3` → **左子树**
- 4 右边：`5 6 7` → **右子树**

现在树长这样：

```c++
        4
      /   \
  [1,2,3] [5,6,7]
```

---

## 第二步：建左子树（1,2,3）

左子树有 3 个节点。

在后序里，前 3 个就是左子树的后序：`2 3 ①`

后序最后一个：**1**

→ **左子树根是 1**

看中序：

`① 2 3 | 4 | ...`

- 1 左边：无
- 1 右边：`2 3` → 1 的右子树

现在树：

plaintext

```c++
         4
       /   \
      1    [5,6,7]
       \
       [2,3]
```

---

## 第三步：建 1 的右子树（2,3）

这部分在后序里是：`2 ③`

最后一个是 **3**

→ 根是 3

看中序：

`1 | ② ③ | ...`

- 3 左边：2
- 3 右边：无

树更新：

plaintext

```c++
         4
       /   \
      1     6
       \   / \
        3 5   7
       /
      2
```

---

## 第四步：建右子树（5,6,7）

右子树共 3 个节点，在后序里是：`5 7 ⑥`

最后一个是 **6**

→ 右子树根是 6

看中序：

`... | 4 | ⑤ ⑥ ⑦`

- 6 左边：5
- 6 右边：7

最终完整树：

plaintext

```c++
         4
       /   \
      1     6
       \   / \
        3 5   7
       /
      2
```

---

## 第五步：层序遍历（从上到下、从左到右）

一层一层数：

1. 第 0 层：4
2. 第 1 层：1, 6
3. 第 2 层：3, 5, 7
4. 第 3 层：2

连起来：

**4 1 6 3 5 7 2**

**结构体写法：（建树）
```c++
#include <iostream>
#include <vector>
#include <queue>
using namespace std;

// 二叉树节点结构
struct Node {
    int val;
    Node *left, *right;
    Node(int v) : val(v), left(NULL), right(NULL) {}
};

vector<int> post, in;
int n;

// 递归建树：后序[pl,pr]  中序[il,ir]
Node* build(int pl, int pr, int il, int ir) {
    if (pl > pr) return NULL;

    // 后序最后一个是根
    int root_val = post[pr];
    Node* root = new Node(root_val);

    // 找到中序里根的位置
    int k;
    for (k = il; k <= ir; k++) {
        if (in[k] == root_val) break;
    }

    int left_num = k - il;

    // 左子树、右子树
    root->left = build(pl, pl + left_num - 1, il, k - 1);
    root->right = build(pl + left_num, pr - 1, k + 1, ir);
/*
后序：`2 3 1 5 7 6 ④`
中序：`1 2 3 ④ 5 6 7`
*/
    return root;
}

// 层序遍历 BFS
void levelOrder(Node* root) {
    queue<Node*> q;
    q.push(root);
    bool first = true;

    while (!q.empty()) {
        Node* cur = q.front();
        q.pop();

        if (!first) cout << " ";
        first = false;
        cout << cur->val;

        if (cur->left) q.push(cur->left);
        if (cur->right) q.push(cur->right);
    }
}

int main() {
    cin >> n;
    post.resize(n);
    in.resize(n);

    for (int i = 0; i < n; i++) cin >> post[i];
    for (int i = 0; i < n; i++) cin >> in[i];

    Node* root = build(0, n - 1, 0, n - 1);
    levelOrder(root);

    return 0;
}
```
**其实这里发现，我们已经把这棵树 构造出来了！，所以 其 前序 | 中序 | 后序 遍历易得**

**用map表模拟建树**：
# 注意：
**用数组模拟很容易炸；因为：数组有很多位置是空信息
(除非是完全二叉树)；
如：n = 30，数组就要开 2 ^ 30 的量，~ `1e10`，肯定炸

解决办法：
所以在完全二叉树中，用数组模拟是完全没有问题；
但是如果是这种题的情况，
1.把int* 换成 map
2.结构体（如上）

```c++
#include <bits/stdc++.h>
using namespace std;

map<int,int> tree;
void build(int post[],int in[],int len,int u) {
    if(len == 0) return;
    tree[u] = post[len - 1];
    if(len == 1) return;
    //find根节点在中序中的位置
    int pos = 0;
    while(in[pos] != post[len - 1]) pos++;
    int left_len = pos;
    int right_len = len - 1 - left_len;
    build(post,in,left_len,2 * u + 1);
    build(post + left_len,in + pos + 1,right_len,2 * u + 2);

bool bfs(int n) {
    queue<int> q;
    q.push(0);
    int cnt = 1;
    while(!q.empty()) {
        int len = q.size();
        for(int i = 0;i < len;i++) {
            int node = q.front();
            q.pop();
            if(tree[node] % cnt != 0) return false;

            if(tree[2 * node + 1] != 0) q.push(2 * node + 1);
            if(tree[2 * node + 2] != 0) q.push(2 * node + 2);
        }
        cnt++;
    }
    return true;
}

        for(int i = 0;i < n;i++) cin >> in[i];
```

```c++
#include <bits/stdc++.h>
using namespace std;

void p(const vector<int>& a,int i,int n,vector<int>& ans) {
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    if(left <= n) p(a,left,n,ans);
    if(right <= n) p(a,right,n,ans);
    ans.push_back(a[i]);
}

bool dfs2(const vector<int>& a,int i,int n) {
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    if(left <= n) {
        if(a[left] < a[i] || !dfs2(a,left,n)) return false;
    }
    if(right <= n) {
       if(a[right] < a[i] || !dfs2(a,right,n)) return false;
    }
    return true;
}

bool dfs1(const vector<int>& a,int i,int n) {
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    if(left <= n) {
        if(a[left] > a[i] || !dfs1(a,left,n)) return false;
    }
    if(right <= n) {
       if(a[right] > a[i] || !dfs1(a,right,n)) return false;
    }
    return true;
}

int main() {
    int T,n;
    cin >> T >> n;
    while(T--) {
        vector<int> a(n);
        for(int i = 0;i < n;i++) cin >> a[i];

        bool vis1,vis2;
        vis1 = dfs1(a,0,n - 1);
        vis2 = dfs2(a,0,n - 1);

        if(!vis1 && !vis2) {
            cout << "Not Heap" << endl;
        } else if(vis1 && !vis2) {
            cout << "Max Heap" << endl;
        } else if(!vis1 && vis2) {
            cout << "Min Heap" << endl;
        }

        vector<int> ans;
        p(a,0,n - 1,ans);
        for(int i = 0;i < ans.size();i++) {
            cout << ans[i];
            if(i != ans.size() - 1) cout << " ";
        }
        if(T != 0) cout << endl;
    }
    return 0;
}
```
