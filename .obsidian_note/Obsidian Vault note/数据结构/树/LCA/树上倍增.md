vector<int> deep;
vector<int> dis;//( + 权值树)
vector<vector<int>> pa;

void dfs(const vector<int>& graph,int x,int fa) {
	pa[0][x] = fa;
	for(auto& [next,w] : graph[x]) {
		if(next != fa) {
			deep[next] = deep[x] + 1;
			dis[next] = dis[x] + w;
			dfs(graph,next,x);
		}
	}
}

//获取对于一个点的第k个祖先点
int get_kth_ancestor(int node,int k) {
	for(;k > 0 && node >= 0;k &= (k - 1)) {
		node = pa[__builtin_ctz(k)][node];
	}
	return node;
}

int get_lca(int x,int y) {
	if(deep[x] > deep[y]) swap(x,y);
	
	//将x 和 y调到同一深度
	y = get_kth_ancestor(y,depth[y] - depth[x]);
	if(y == x) return x; 
	
	for(int i = pa.size() - 1;i >= 0;i--) {
		int px = pa[i][x],py = pa[i][y];
		if(px != py) {
			x = px;
			y = py;
		}
	}
	return pa[0][x];
}

//获取点x到y的最短距离
long long get_dis(int x ,int y) {
	return dis[x] + dis[y] - 2 * dis[get_lca(x,y)]; 
}
