# 对于区间的极值问题

## 1.按顺序给定区间的极值

![配图](../../../截图文件/Pasted%20image%2020260420161514.png)

在滑窗的思想上，可以用map表来存值，因为map中的底层适配器是红黑树,key(键)是按照升序的规则存储
区间的极值就在`mp.begin()->first`(min) / `mp.rbegin()->first`(max)上；

**本题是一个定长滑窗
```c++
#include <bits/stdc++.h>
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(0);
    
    int n,k;
    cin >> n >> k;
    vector<int> a(n);
    for(int i = 0;i < n;i++) cin >> a[i];
    
    //初始窗口
    map<int,int> mp;
    for(int i = 0;i < k;i++) {
        mp[a[i]]++;
    }
    //max
    auto it = mp.rbegin();
    cout << it->first << " ";
    
    //滑动窗口
    for(int i = 1;i <= n - k;i++) {
        int l = a[i - 1];
        if(--mp[l] == 0) {
            mp.erase(l);
        }
        
        int r = a[i + k - 1];
        mp[r]++;
        //max
        auto it = mp.rbegin();
        cout << it->first;
        if(i != n - k) cout << " ";
    }
    return 0;
}
```




## 2.若对于任意一个区间求极值，也就是无序
方法：
· ST表
· 树状数组；
· 线段树；

**但是这里有些区别，ST表only can solve such like static problem ,but 线段树 和 树状数组
not only can solve it,but also can solve dynamic problem.
也就是说 树状数组 可以完成对于单点区间的更新，同时对任意区间所维护数据的查询；具体的看我obsdian 数据结构->树状数组；

## 例题：
![配图](../../../截图文件/Pasted%20image%2020260420164140.png)

**ST 表
```c++
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int mod = 1e9 + 7;
const int N = 3e5 + 7;

void solve() { 
    int m,n;
    cin >> m >> n;
    int max_j = log2(m) + 1;
    vector<vector<int>> st(m,vector<int>(max_j));
    for(int i = 0;i < m;i++) {
        cin >> st[i][0]; 
    }
    for(int j = 1;j < max_j;j++) {
        for(int i = 0;i + (1 << j) <= m;i++) {
            st[i][j] = min(st[i][j - 1],st[i + (1 << (j - 1))][j - 1]); 
        }
    }
    while(n--) {
        int a,b;
        cin >> a >> b;
        int len = b - a + 1;
        a--;
        b--;
        int k = log2(len);
        int ans = min(st[a][k],st[b - (1 << k) + 1][k]);
        cout << ans << " ";
    }
    return;
}
signed main() {
    ios::sync_with_stdio(false);    
    cin.tie(nullptr);    
    cout.tie(0);
    int T = 1;
    //cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

**树状数组解法：
```c++
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int mod = 1e9 + 7;
const int N = 3e5 + 7;
const int INF =1e9 + 10;

vector<int> tree;
vector<int> a;

void init(int n) {
    tree.resize(n + 1);
    for(int i = 1;i <= n;i++) tree[i] = INF;    
}
void update(int a,int idx,int n) {
    for(int i = idx;i <= n;i += i & -i) {
        tree[i] = min(tree[i],a);
    }
}
int query(int l,int r) {
    int mi = INF;
    while(r >= l) {
        int next = r & (r - 1);
        if(next >= l) {
            mi = min(mi,tree[r]);
            r = next;
        }
        else {
            mi = min(a[r],mi);
            r--;
        }
    }
    return mi;
}
void solve() {
    int n,k;
    cin >> n >> k;
    a.resize(n + 1);
    init(n);
    for(int i = 1;i <= n;i++) {
        int nums;
        cin >> nums;
        a[i] = nums;
        update(nums,i,n);
    }
    while(k--) {
        int l,r;
        cin >> l >> r;
        cout << query(l,r) << " ";
    }
}
signed main() {
    ios::sync_with_stdio(false);    
    cin.tie(nullptr);    
    cout.tie(0);
    int T = 1;
    //cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
