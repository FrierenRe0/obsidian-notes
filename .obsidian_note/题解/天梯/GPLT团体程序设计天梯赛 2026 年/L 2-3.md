解读：
对于本题考察点在于图论中的单向路的路径搜索问题：
本题是在求出在每条路径中的最小权值的最大值，同时记录最小权值为最大值的路径的叶子节点；

**思考方向**：
**1.单向图的搜索问题的首先可以采用 `dfs`把每条路径深搜，维护最小权值，在记录最小权值为最大值的路径的叶子节点**

```c++
#include <bits/stdc++.h>
using namespace std;
//map<int,int> mp;
vector<int> ans;
int mx = INT_MIN;
void dfs(const vector<vector<pair<int,int>>>& graph,int u,int mi) {
    //终止条件
    if(graph[u].size() == 0) {
        if(mx < mi) {
            ans = {u};
            mx = mi;
        }
        else if(mx == mi) ans.push_back(u);
        return;
    }
    //遍历深搜
    for(auto [next,f] : graph[u]) {
        dfs(graph,next,min(f,mi));
    }
}
int main() {
    int n;
    cin >> n;

    vector<vector<pair<int,int>>> graph(n);
    //有向图
    for(int i = 1;i <= n - 1;i++) {
        int u,f;
        cin >> u >> f;
        graph[u].push_back({i,f});
    }

    dfs(graph,0,INT_MAX);
    //输出
    cout << mx << endl;
    sort(ans.begin(),ans.end());
    for(int i = 0;i < ans.size();i++) {
        cout << ans[i];
        if(i != ans.size() - 1) cout << " ";
    }

    return 0;
}
```

## 注意:

**这里主播犯了个我之前很少看到的错误：❌
用两个全局map记录搜索过的权值，mp.begin()->first就是路径上最小权值
这样写的问题在于：
·全局map,在递归回溯时用erase会发生了迭代器失效 / 重复删除！
·mp 是全局变量！
在深度递归时，上层还在遍历，下层却把 key 删掉了
→ 迭代器失效！
→ 直接段错误！

**打个比方**：
你爷爷正在桌子上吃饭（遍历）
你突然从房间里出来，把桌子掀了（erase）
→ 爷爷当场摔倒！
→ 程序崩溃！
这就有可能迭代器会指向空map,导致迭代器失效 / 重复删除 ❌

**（其实完全没必要这么做，是因为模拟赛中写大模拟写昏了`hhh`,用个mi直接调用函数值传递就行）**

## 错误写法（不要请以模仿）❌
```c++
map<int,int> mp; ❌
vector<int> ans;
int mx = INT_MIN;
void dfs(const vector<vector<pair<int,int>>>& graph,int u) {
	//终止条件
	if(graph[u].size() == 0) {
		auto it = mp.begin();
		int mi = it->first;
		if(mx < mi) {
			ans = {u};
			mx = mi;
		}
		else if(mx == mi) ans.push_back(u);
		return;
	}
	//遍历深搜
	for(auto [next,f] : graph[u]) {
		mp[f]++;
		dfs(graph,next);
		//回溯
		if(--mp[f] == 0) {
			mp.erase(f);  ❌
		}
	}
}
```


## **2.因为是单向图搜索问题，可以采用`dijk`算法
（模板中是用dis数组来记录根节点到每一个节点上的最短路（单源最短路），这里我们将dis中记录的路径sum,换成路径中遍历过的最小权值即可）
```c++
#include <bits/stdc++.h>
using namespace std;
vector<int> ans;
int mx = INT_MIN;

int main() {
    int n;
    cin >> n;

    //有向图
    vector<vector<pair<int,int>>> graph(n);
    for(int i = 1;i <= n - 1;i++) {
        int u,f;
        cin >> u >> f;
        graph[u].push_back({i,f});
    }

	//dijk
    vector<int> dis(n,INT_MAX);
    vector<bool> vis(n,false);
    priority_queue<pair<int,int>,vector<pair<int,int>>,greater<>> pq;
    pq.push({INT_MAX,0});
    while(!pq.empty()) {
        auto [f,u] = pq.top();
        pq.pop();
        if(dis[u] < f) continue;
        else vis[u] = true;
        for(auto [next,w] : graph[u]) {
            int mi = min(w,f);
            if(dis[next] > mi) {
                dis[next] = mi;
                pq.push({mi,next});
            }
        }
    }

    for(int i = 0;i < n;i++) {
        if(graph[i].size()) continue;
        if(dis[i] > mx) {
            mx = dis[i];
            ans = {i};
        } else if(dis[i] == mx) ans.push_back(i);
    }
    //输出
    cout << mx << endl;
    for(int i = 0;i < ans.size();i++) {
        cout << ans[i];
        if(i != ans.size() - 1) cout << " ";
    }
    return 0;
}
```
