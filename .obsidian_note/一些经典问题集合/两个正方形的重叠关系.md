[D-小红的正方形_牛客周赛 Round 150](https://ac.nowcoder.com/acm/contest/137073/D)

**题目大况**：
**给定两个正方形的两个对角线顶点从而唯一确定，再问你两个正方形的重叠性。如果两者的边界有无穷多个公共点，请输出 inf。**

**思路**：
**这里思路很明确，但是代码写起来有点屎；
如果出现重叠（inf），只有当两个正方形的边出现边与边的重叠才有可能；
所以我们可以先看给定的两个正方形是否存在重边情况，
- **若有输出inf,结束
- **否则，在看边和边的交叉情况，遍历一个正方形的各个边，看与另一个正方形各个边的交叉情况。
- **这里需要注意的是重复性问题，pair<int，int> 表示的是这两条边的横纵情况，p.first 为一对交叉线的纵线的x坐标，p.second为一对交叉线为横线的y坐标；当出现重复的pair时，这里转化成图的形式就是两个正方形顶点对顶点的情况，所以这里需要我们去完成去重（set<pair<int,int>>即可）；

**代码如下**：
```c++
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int INF = 1e18;

bool segOverlap(int a, int b, int c, int d) {
    int l = max(a, c);
    int r = min(b, d);
    return l < r;
}

bool isInf(int L1, int R1, int B1, int T1, int L2, int R2, int B2, int T2) {
    if (L1 == L2 && segOverlap(B1, T1, B2, T2)) return true;
    if (L1 == R2 && segOverlap(B1, T1, B2, T2)) return true;
    if (R1 == L2 && segOverlap(B1, T1, B2, T2)) return true;
    if (R1 == R2 && segOverlap(B1, T1, B2, T2)) return true;

    if (B1 == B2 && segOverlap(L1, R1, L2, R2)) return true;
    if (B1 == T2 && segOverlap(L1, R1, L2, R2)) return true;
    if (T1 == B2 && segOverlap(L1, R1, L2, R2)) return true;
    if (T1 == T2 && segOverlap(L1, R1, L2, R2)) return true;
    return false;
}

pair<int, int> cross(int hy, int hx1, int hx2, int vx, int vy1, int vy2) {
    if (vx >= hx1 && vx <= hx2 && hy >= vy1 && hy <= vy2) {
        return {vx, hy};
    }
    return {INF, INF};
}

void solve() {
    int x1, y1, x2, y2;
    cin >> x1 >> y1 >> x2 >> y2;
    //这里L,R是表示两个纵线的x坐标(一左一右)
    int L1 = min(x1, x2), R1 = max(x1, x2);
    //这里B,T是表示两个横线的y坐标(一下一上)下面同理
    int B1 = min(y1, y2), T1 = max(y1, y2);

    cin >> x1 >> y1 >> x2 >> y2;
    int L2 = min(x1, x2), R2 = max(x1, x2);
    int B2 = min(y1, y2), T2 = max(y1, y2);

	//这里则是判断是否存在重边
    if (isInf(L1, R1, B1, T1, L2, R2, B2, T2)) {
        cout << "inf\n";
        return;
    }

	//如果不存在，在计算一共有多少一个不重叠的交叉组，
    set<pair<int, int>> st;
    vector<tuple<char, int, int, int>> edgeA;
    //v -> 表示是纵线， h -> 表示是横线
    //v中，第2个数据是表示纵线的x坐标，第3，4个是这个纵线的两个端点的y坐标
    //h中，第2个数据是表示横线的y坐标，第3，4个是这个横线的两个端点的x坐标
    edgeA.emplace_back('v', L1, B1, T1);
    edgeA.emplace_back('v', R1, B1, T1);
    edgeA.emplace_back('h', B1, L1, R1);
    edgeA.emplace_back('h', T1, L1, R1);

    vector<tuple<char, int, int, int>> edgeB;
    edgeB.emplace_back('v', L2, B2, T2);
    edgeB.emplace_back('v', R2, B2, T2);
    edgeB.emplace_back('h', B2, L2, R2);
    edgeB.emplace_back('h', T2, L2, R2);

	//遍历统计
    for (auto &ea : edgeA) {
        char ta = get<0>(ea);
        int a1 = get<1>(ea), a2 = get<2>(ea), a3 = get<3>(ea);
        for (auto &eb : edgeB) {
            char tb = get<0>(eb);
            int b1 = get<1>(eb), b2 = get<2>(eb), b3 = get<3>(eb);
            if (ta == tb) continue;
            pair<int, int> p;
            if (ta == 'h' && tb == 'v') {
                p = cross(a1, a2, a3, b1, b2, b3);
            } else {
                p = cross(b1, b2, b3, a1, a2, a3);
            }
            if (p.first != INF) st.insert(p);
        }
    }
    //输出
    cout << st.size() << '\n';
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    while (T--) solve();
    return 0;
}
```
